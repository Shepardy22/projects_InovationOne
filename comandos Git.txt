comandos basicos Git

windows

cd (para navegar para pastas especificas)
dir (listar as pastas do diretorio)
mkdir (criar pastas nos diretorios)
del / rmdir (deletar arquivos de uma pastas) / rmdir pastaxyz /S /Q (deletar pasta completa)
echo (criar arquivos ex. echo arquivo > arquivo.txt)
openssl sha1 arquivo.txt 


unix/linux

echo (criar arquivos ex. echo arquivo > arquivo.txt)
cd (para navegar para pastas especificas)
ls (listar as pastas do diretorio)
mkdir (criar pastas nos diretorios)
del / rmdir (deletar arquivos de uma pastas) / rmdir pastaxyz /S /Q (deletar pasta completa)
rm / -rf
